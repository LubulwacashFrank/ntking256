function sendMessage() {
    const input = document.getElementById('messageInput');
    const message = input.value.trim();
    
    if (message) {
        addMessage(message, true);
        input.value = '';
        
        // Simulate response after random delay
        setTimeout(() => {
            const responses = [
                "Thank you for your interest! Let me check availability and get back to you shortly.",
                "That sounds good! Let me confirm the details.",
                "Great! I have that available. Would you like to proceed?",
                "I understand. Let me prepare a quote for you."
            ];
            const randomResponse = responses[Math.floor(Math.random() * responses.length)];
            addMessage(randomResponse, false);
        }, 1000 + Math.random() * 1500);
    }
}

function addMessage(text, isSent) {
    const messagesArea = document.getElementById('messagesArea');
    const now = new Date();
    const time = now.toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit' });
    
    const messageHTML = `
        <div class="message ${isSent ? 'message-sent' : 'message-received'}">
            ${!isSent ? `<div class="message-sender">${document.getElementById('chatUserName').textContent}</div>` : ''}
            <div class="message-text">${text}</div>
            <div class="message-time">${time}</div>
        </div>
    `;
    
    messagesArea.insertAdjacentHTML('beforeend', messageHTML);
    messagesArea.scrollTop = messagesArea.scrollHeight;
}

function handleMessageKey(event) {
    if (event.key === 'Enter' && !event.shiftKey) {
        event.preventDefault();
        sendMessage();
    }
}

function updatePrices() {
    livePrices.forEach((price, index) => {
        const change = Math.floor(Math.random() * 20) - 10;
        price.price = Math.max(500, price.price + change);
        price.change = Math.abs(change);
        price.trend = change >= 0 ? 'up' : 'down';
    });
    
    renderLivePrices();
    
    const now = new Date();
    document.getElementById('lastUpdateTime').textContent = now.toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit' });
}

function startActivityCounters() {
    setInterval(() => {
        // Simulate changing numbers
        const farmersEl = document.getElementById('farmersCount');
        const buyersEl = document.getElementById('buyersCount');
        const dealsEl = document.getElementById('dealsCount');
        const chatsEl = document.getElementById('chatsCount');
        
        if (farmersEl) {
            let count = parseInt(farmersEl.textContent.replace(',', ''));
            count += Math.floor(Math.random() * 3) - 1;
            farmersEl.textContent = count.toLocaleString();
        }
        
        if (buyersEl) {
            let count = parseInt(buyersEl.textContent.replace(',', ''));
            count += Math.floor(Math.random() * 10) - 4;
            buyersEl.textContent = count.toLocaleString();
        }
        
        if (dealsEl) {
            let count = parseInt(dealsEl.textContent);
            count += Math.floor(Math.random() * 3);
            dealsEl.textContent = count;
        }
        
        if (chatsEl) {
            let count = parseInt(chatsEl.textContent);
            count += Math.floor(Math.random() * 2);
            chatsEl.textContent = count;
        }
    }, 5000);
}
